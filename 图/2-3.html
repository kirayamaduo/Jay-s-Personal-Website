☕ Java Object Layer
User.java
@Entity
@Table(name="users")
class User {
  @Id Long id;
  String username;
  @OneToMany
  List<Post> posts;
}
Post.java
@Entity
class Post {
  @Id Long id;
  String title;
  @ManyToOne
  @JoinColumn("user_id")
  User author;
}
